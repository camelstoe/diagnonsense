First time setup — 5 minutes

1Create a Google Sheet
Go to sheets.new and name it "Project Diagnonsense".
2Open Apps Script
In the sheet: click Extensions → Apps Script. Delete everything in the editor.
3Paste the script
Copy the code below and paste it into Apps Script, then click the save icon.
const SHEET_NAME = "Entries"; function doPost(e) { const lock = LockService.getScriptLock(); lock.tryLock(10000); try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow([ "Timestamp","Date","Time slot","Temp control (°C)", "Diff L knee","Diff R knee","Diff L elbow","Diff R elbow", "Diff L hand","Diff R hand","Diff L jaw","Diff R jaw", "Raw L knee","Raw R knee","Raw L elbow","Raw R elbow", "Raw L hand","Raw R hand","Raw L jaw","Raw R jaw", "Circ L knee (mm)","Circ R knee (mm)","Circ L elbow (mm)","Circ R elbow (mm)", "Thumb L","Thumb R","Index L","Index R","Middle L","Middle R","Ring L","Ring R","Little L","Little R","Jaw opening (mm)", "Pain head","Pain neck","Pain hips","Pain knees","Pain elbows","Pain hands", "Pain jaw","Pain feet","Pain fatigue", "Sun exposure","UV index","Hours since sun","Skin coverage", "Swelling visible","Photos taken of","Pain types","Head location","Head type","Head area","Ice wanted","Ice applied","Medications","Notes" ]); sheet.setFrozenRows(1); } const d = JSON.parse(e.postData.contents); sheet.appendRow([ new Date().toISOString(), d.date, d.time, d.tempControl, d.dLknee, d.dRknee, d.dLelbow, d.dRelbow, d.dLhand, d.dRhand, d.dLjaw, d.dRjaw, d.rLknee, d.rRknee, d.rLelbow, d.rRelbow, d.rLhand, d.rRhand, d.rLjaw, d.rRjaw, d.cLknee, d.cRknee, d.cLelbow, d.cRelbow, d.cLthumb, d.cRthumb, d.cLindex, d.cRindex, d.cLmiddle, d.cRmiddle, d.cLring, d.cRring, d.cLlittle, d.cRlittle, d.cJaw, d.pHead, d.pNeck, d.pHips, d.pKnees, d.pElbows, d.pHands, d.pJaw, d.pFeet, d.pFatigue, d.sunExp, d.uvIndex, d.sunHours, d.sunCoverage, d.swelling, d.photos, d.painTypes, d.headLoc, d.headType, d.headArea, d.iceWanted, d.iceApplied, d.meds, d.notes ]); return buildResponse({status:"ok"}); } catch(err) { return buildResponse({status:"error", message: err.toString()}); } finally { lock.releaseLock(); } } function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return buildResponse({rows:[]}); const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = data.slice(1).map(r => { const obj = {}; headers.forEach((h,i) => obj[h] = r[i]); return obj; }); return buildResponse({rows}); } catch(err) { return buildResponse({status:"error", message: err.toString()}); } } function buildResponse(data) { const output = ContentService.createTextOutput(JSON.stringify(data)); output.setMimeType(ContentService.MimeType.JSON); return output; }
